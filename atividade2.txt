Algoritmo_Carro
var: custoFabrica, impostos, distribuidor, custoFinal, subtotal
Leia ("Digite o custo de fábrica do carro: "), custoFabrica;
impostos <- custoFabrica * 0.45;
subtotal <- custoFabrica + impostos;
distribuidor <- subtotal * 0.28;
custoFinal <- subtotal + distribuidor;
Escreva ("Custo ao consumidor: "), custoFinal;
Fim
